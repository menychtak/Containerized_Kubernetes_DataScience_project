document.getElementById('addTaskBtn').addEventListener('click', () => { const task = prompt("Enter a new task:"); if (task) { fetch('http://backend:5000/tasks/add', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ task: task }) }) .then(response => response.json()) .then(data => { if (data.message === 'Task added successfully') { const li = document.createElement('li'); li.textContent = task; li.addEventListener('click', () => { li.classList.toggle('completed'); }); document.getElementById('tasks').appendChild(li); } }); } });
